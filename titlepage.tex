	\end{figure}
	
	\vspace{\stretch{1}}
	\begin{LARGE}
	
	\par\noindent%
	 \IfStrEqCase{\cmdKV@thesisTitlePage@type}{%
	  {sem}{SEMINARARBEIT}%

	  }[\cmdKV@thesisTitlePage@type]
